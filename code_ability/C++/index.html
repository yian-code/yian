    cin >> ch;
    cout << "Enter an integer: ";

    char *ps = buildstr(ch, times);

    cout << ps << "-DONE-" << ps << endl;
    delete [] ps;

char * buildstr(char c, int n)
{
    char * pstr = new char[n + 1];
    //从后向前填充,避免了使用额外的变量
    // int i = 0;
    // while (i < n)
    // {
    //     pstr[i++] = c;/* code */
    // } 
    pstr[n] = '\0';
    while (n-- > 0)
        pstr[n] = c;
    return pstr;
}
unsigned int c_in_str(const char * str, char ch)
{
    unsigned int count = 0;
    // 处理字符串中字符的标准方式,使用地址遍历而不是另外使用for循环
    while (*str)
    {
        if (*str == ch)
            count++;
        str++;
    }
    return count;
}

#include <iostream>
int main()
{
    using namespace std;
    double wages[3] = {10000.0, 20000.0, 30000.0};
    short stack[3] = {3, 2, 1};

    double * pw = wages;
    short * ps = &stack[0];

    cout << "pw = " << pw <<", *pw = " << *pw << endl;
    pw += 1;
    cout <<"add 1 to the pw pointer:\n";
    cout << "pw = " << pw <<", *pw = " << *pw << "\n\n";
    cout << "ps = " << ps <<", *ps = " << *ps << endl;
    ps += 1;
    cout <<"add 1 to the ps pointer:\n";
    cout << "ps = " << ps <<", *ps = " << *ps << "\n\n";

    cout << "stack[0] = " << stack[0]
            <<", stack[1] = " << stack[1] << "\n";
    
    cout << "*stack = " << *stack
        <<", *(stack + 1) = " << *(stack + 1) << endl;
    cout << sizeof(wages) << "= size of wages array\n";
    cout << sizeof(pw) << "size of pw pointer\n";
    return 0;
}

# 递增与递减运算符

++a与a++的区别：简单的讲a++是先进行运算再自增,++a是先自增再参与运算

#include <iostream>
int main()
{
    using std::cout;
    int a = 20;
    int b = 20;
    cout << "a  = " << a << ":  b = " << b << "\n";
    cout << "a++  = " << a++ << ":  ++b = " << ++ b << "\n";
    cout << "a  = " << a << ":  b = " << b << "\n";
    return 0;
}
/* 
a  = 20:  b = 20
a++  = 20:  ++b = 21
a  = 21:  b = 21
 */

# 编写延时函数

long wait = 0;
while (wait < 10000)
    wait++;

头文件ctime中提供了解决策略,首先它定义了一个符号常量CLOCKS_PER_SEC,该常量等于每秒钟包含的系统时间单位数,因此将系统时间除以这个值或者将秒数乘以CLOCKS_PER_SEC,可以得到以系统时间为单位的时间,其次ctime将clock_t作为clock()返回类型的别名:以下演示使用clock()创建的延时函数

#include <iostream>
#include <ctime>
int main()
{
    using namespace std;
    cout << "Enter the delay time, in seconds: ";
    float secs;
    cin >> secs;
    clock_t delay = secs * CLOCKS_PER_SEC;
    cout << "staring\a\n";
    clock_t start = clock();
    while (clock() - start < delay)
        ;
    cout << "done \a\n";
    return 0;
}

# 使用clock()计时

clock()open in new window 函数返回从“开启这个程序进程”到“程序中调用clock()函数”时之间的CPU时钟计时单元(clock tick)数

# eg:

#include<iostream>
#include<cstdlib>
#include<ctime>
using namespace std;
clock_t start,end;
 
int main(){
	start=clock();		//程序开始计时
	int ans=0;
	for(int i=1;i<=1e8;i++)
		ans++;
	end=clock();		//程序结束用时
	double endtime=(double)(end-start)/CLOCKS_PER_SEC;
	cout<<"Total time:"<<endtime<<endl;		//s为单位
	cout<<"Total time:"<<endtime*1000<<"ms"<<endl;	//ms为单位
	system("pause");
	return 0;
}

# 基于范围的for循环(range-based)

简化了一种常见的循环任务:对数组(或容器类,如vector和array)的每个元素执行相同的操作:

#include <iostream>
int main()
{
    double price[5] = {4.2, 3.4, 4.5, 6.5, 5.7};
    for (double x: price)
        std::cout << x << std::endl;
}
/* 4.2
3.4
4.5
6.5
5.7 */




 







但是如果要修改数组中的元素,需要使用不同的循环变量的语法:

for (double &x : price)
// 符号&表示x是一个引用变量,这样才能修改数组,而上面的代码不能
    x = x * 0.80;
 


还可以结合使用基于范围的for循环和初始化列表:这种循环主要用于模板容器类

for (int x : {3, 5, 2, 8, 6})
    cout << x << " ";
cout << '\n';
// 3 5 2 8 6
 



# 字符串函数库cctype

cctype函数库可以简化确定字符是否为大写字母、数字、标点符号等工作,这些函数的原型在头文件cctype中定义的

函数名称返回值
isalnum()如果括号内参数是字母或数字,函数就返回true,否则返回false
isalpha()如果参数是字母,函数返回true, 否则返回false
islower()如果参数是小写字母,函数返回true,通常我们也可以这样判断小写字母 c >= 'a' && c <= 'z'
isupper()如果参数是大写字母,函数返回true,通常我们也可以这样判断大写字母 c >= 'A' && c <= 'Z'
isdigit()如果参数是数字,函数返回true,通常我们也可以这样判断数字 c >= '0' && c <= '9'
isxdigit()如果参数是十六进制数字,即0 ~ 9,a ~ f或A ~ F,函数返回true
ispunct()如果参数是标点符号,函数返回true
iscntrl()如果参数是控制字符,函数返回true
isgraph()如果参数除空格之外的打印字符,函数返回true
isprint()如果参数是打印字符(包括空格' '),函数返回true
isspace()如果参数是标准空白字符,如空格' '、换行符'\n'、水平制表符'\t'等
tolower()如果参数是字母(无论大小写),函数返回其小写字母,否则返回原字符
toupper()如果参数是字母(无论大小写),函数返回其大写字母,否则返回原字符

TIP

  • 什么是控制字符？ 控制字符也叫不可见字符,即ASCII码中不能正确打印显示出来的字符
  • 什么是打印字符？ 打印字符也叫可见字符,即ASCII码中可以正确打印显示出来的字符

# 读取文本文件

#include <iostream>
#include <fstream>
#include <cstdlib> //support exit()
const int SIZE = 60;

int main()
{
    using namespace std;
    char filename[SIZE];
    ifstream inFile;//创建文件对象
    cout << "Enter name of data file: ";
    cin.getline(filename, SIZE);
    inFile.open(filename);
    if (!inFile.is_open())
    {
        cout << "Could not open the file" << filename << endl;
        cout << "Progream terminating.\n";
        exit(EXIT_FAILURE);
    }
    double value;
    double sum = 0.0;
    int count = 0;

    inFile >> value;//传入第一个value
    while (inFile.good())
    {
        ++count;
        sum += value;
        inFile >> value;
    }
    if (inFile.eof())//判断是否读取完
        cout << "end of file reached\n";
    else if (inFile.fail())//读取失败
        cout << "Input terminated by data mismatch.\n";
    else
        cout << "Input terminated for unknown reason\n";
    if (count == 0)
        cout << "No data processed\n";
    else
    {
        cout << "Item read: " << count << endl;
        cout << "Sum: " << sum << endl;
        cout << "Average: " << sum / count << endl;
    }
    inFile.close();
    return 0;
}

/*
//data.txt

15 34 19.4
17.5

Enter name of data file: data.txt
end of file reached
Item read: 7
Sum: 145.1
Average: 20.7286
*/

  1. 函数原型:描述了函数到编译器的接口,将函数返回值的类型(如果有的话)以及参数的类型和数量告诉编译器

  3. 原型功能:确保编译器正确处理函数返回值、编译器检查使用的参数目录是否正确、编译器检查使用参数类型是否正确,不正确则转换为正确的类型(如果可能的话)

通常在原型参数列表中可以包含变量名,也可以不包含.原型中的变量名相当于占位符,因此不必与函数定义中的变量名相同

# 函数和数组

cookies是数组名,前面介绍过,C++将数组名解释为其第一个元素的地址:

cookies == &cookies[0]

当我们定义函数int sum = sum_arr(int arr[], int n)时候真正的函数头应该是这样的:

int sum = sum_arr(int * arr, int n)

将数组作为函数的参数意味着:程序实际并没有将数组内容传递给函数,而是将数组的地址,包含元素种类和元素数目提供给函数

#include <iostream>
const int ArSize = 8;
int sum_arr(int arr[], int n);
int main()
{
    int cookies[ArSize] = {1, 2, 4, 8, 16, 32, 64, 128};
    std::cout << cookies << " = array address, ";
    std::cout << sizeof cookies << " = sizeof cookies\n";
    int sum = sum_arr(cookies, ArSize);
    std::cout << "Total cookies eaten: " << sum << std::endl;
    // 前面三个元素的和,提供了一个假的元素个数
    sum = sum_arr(cookies, 3);
    std::cout << "First three eaters ate " << sum <<"cookies\n"; 
    //后面四个元素的和,提供了一个假的起始位和假的元素个数    
    // sum = sum_arr(cookies + 4, 4);
    sum = sum_arr(&cookies[4], 4);//与上一句含义相同
    std::cout << "Last four eaters ate" << sum << " cookies\n";
    return 0;
}
int sum_arr(int arr[], int n)
{
    int total = 0;
    std::cout << arr << " = arr";
    std::cout << sizeof arr << " = sizeof arr\n";
    for (int i = 0; i <n; i++)
    {
        total = total + arr[i];
    }
    return total;
}

# 递归

递归:函数自己调用自己,使用断点调试可以轻松看出递归运行的流程

void recurs(argumentlist)
{
    statements1
    if (test)
        recurs(arguments)
    statements2
}

有趣的是:if为true时,每个recurs()调用都将执行statements1,然后再调用recurs(),而不会执行statements2,当if为false时,当前调用会执行statements2,执行后程序控制权将返回给调用它的recurs(),以此类推.则当statements1执行完后,statements2将以函数调用相反的顺序执行,以下程序演示了这种行为

#include <iostream>
void countdown(int n);
int main()
{
    countdown(4);
    return 0;
}
void countdown(int n)
{
    using namespace std;
    cout << "Counting down ... " << n << endl;
    if (n > 0)
        countdown(n-1);
    cout << n << ": Kaboom!\n";
}
