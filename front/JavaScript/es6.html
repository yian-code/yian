# var、const、let

ES5只有两种声明变量的方式:var和function,ES6除了添加了let和const,还添加了import与class命令,所以ES6一共有六种声明变量的方法

编程风格

  • 建议不再使用var命令,而是使用let命令取代.
  • let与const之间优先使用const,尤其是在全局环境,不应该设置变量,只应设置常量,有利于提高程序运行效率
  • 所有的函数都应该设置为常量

# const关键字

常量是块级范围的,非常类似用let语句定义的变量.但常量的值是无法(通过重新赋值)改变的,也不能被重新声明,本人理解应该是与C++等语言中作用是一样的

  • 一旦声明必须赋值,不能使用null占位
  • 如果值是基本数据类型,则值无法被改变
  • 如果声明的是引用数据类型,则地址无法改变,但值可以修改
{
    const obj = { age: 20 }
    obj.age = 30;  // 这是允许的
}

一个常量不能和它所在作用域内的其他变量或函数拥有相同的名称

# var关键字与let关键字

  • var语句用于声明一个函数范围或全局范围的变量,无论在何处声明变量,都会在执行任何代码之前进行处理.这被称为变量提升

显式与隐式变量

  • 显式创建:带var的创建方式
  • 隐式创建:不带var的创建方式
  • let声明一个块级作用域的局部变量,并可以初始化为一个值.let value = 1;

let与var的区别:

  • let允许你声明一个作用域被限制在块作用域中的变量、语句或者表达式.暂时性死区
  • var声明的变量作用域是全局或者整个函数块的.不存在死区
for(var a = 1; a < 5; a++){
    setTimeout(() => { console.log('a:', a); }, 0);
    console.log(a);
}
// a: 5
// a: 5
// a: 5
// a: 5

for(let a = 1; a < 5; a++){
    setTimeout(() => { console.log('a:', a); }, 0);
    console.log(a);
}
// a: 1
// a: 2
// a: 3
// a: 4
  • var和let的另一个重要区别,let声明的变量不会在作用域中被提升,也就是必须在使用之前就创建,否则会报错
function fn(){
    console.log(a);
    var a = 10;
}
fn(); // undefined 

function fn(){
    console.log(a);
    let a = 40;
}
fn();
// Uncaught ReferenceError: Cannot access 'a' before initialization
  • 在全局作用域中来看,let不会在顶层对象上创建属性:
var a = 'yian';
let b = 'yian';

console.log(this.a); // yian
console.log(this.b); // undefined

# 解构赋值

ES6中允许按照一定的模式从数组和对象中提取值.解构赋值语法是一种 Javascript 表达式,通过解构赋值,可以将属性/值从对象/数组中取出,赋值给其他变量.

  • 对象解构赋值
  • 数组解构赋值
  • 字符串解构赋值
  • 函数参数的解构赋值
  • 解构赋值的用途

# 对象解构赋值

var personSystemInfoUser = {userName: 'yian', age: 20, sex: '1'}
// var userName = personSystemInfoUser.userName;
// var age = personSystemInfoUser.age;
// var sex = personSystemInfoUser.sex;

var { userName, age, sex } = personSystemInfoUser;
console.log(userName, age, sex);
// yian 20 1

变量名必须和属性名一样,且在声明时候必须同时赋值,如果解构失败,变量的值等于undefined.

var { Name, age, sex } = personSystemInfoUser;
console.log(userName, age, sex);
// Uncaught ReferenceError: userName is not defined

var { userName, age, sex };
{ userName, age, sex } = personSystemInfoUser;
// Uncaught SyntaxError: missing = in destructuring declaration

当变量名与属性名不一致时,必须使用以下写法

var objUser = { userName: 'yian', sex: 1, id:'212121212' }

var { userName:name, sex:sex, id:_id } = objUser;

console.log('name =', name, 'sex =', sex, '_id =', _id);
// name = yian sex = 1 _id = 212121212


 



这也说明了对象的解构赋值机制是先找到同名的属性,然后再赋值给对应的变量,真正被赋值的是变量而不是属性

  • 对象的解构赋值也可以很方便的将现有对象的方法赋到某个变量上
let { log, sin, cos } = Math;

const { log } = console;
log('hello yian') // hello yian
  • 解构也可用于嵌套结构的对象
let obj = { p: [ 'Hello', { y: 'yian' } ] };

let { p: [x, { y }] } = obj;
x // "Hello"
y // "yian"
  • 也可以像数组解构一样赋与默认值.默认值生效的条件是属性值严格等于undefined
  • 如果要使用已经声明好的变量用于解构赋值,需要注意以下情况
var personSystemInfoUser = {userName: 'yian', age: 20, sex: '1'}

// 错误的做法,->大括号在行首会被解析成代码块
var userName, age, sex ;
{ userName, age, sex } = personSystemInfoUser;

// 正确的
var userName, age, sex ;
({ userName, age, sex } = personSystemInfoUser);
console.log(userName, age, sex);

解构赋值的拷贝是浅拷贝.解构赋值拷贝的是这个值的引用,而不是这个值的副本

# 数组解构赋值

变量名可以随意设置,但是解构的内容必须从数组中依次拿取.

var [a, b, c] = [1, 2, 3];
console.log(a, b, c);
// 1 2 3

var [a, ...c] = [1, 2, 3];
console.log(a, c);
// 1 Array [ 2, 3 ]

 



 

只要左右两边的模式相同,就可以进行合法赋值

let [A, [[B], C]] = [10, [[20], 30]];

console.log('A=', A, 'B=', B, 'C=', C);
// A= 10 B= 20 C= 30

解构可以赋默认值

let [name = 'yian'] = ['易安'];
// let [name = 'yian'] = [];
console.log('name=', name);
// name= 易安
// name= yian

不要试图给解构后的变量赋值以改变解构的数组或对象,因为解构是直接等同于使用,而不是简单的赋与地址

var arr = [1, 2, 3]

var [a, b, c] = arr; 
/**  =  
  **** var a = arr[0];
  **** var b = arr[1];
  **** var c = arr[2];
**/
console.log('a =', a);
// a = 1

a = 9;
console.log('a =', a);
// a = 9
console.log(arr);
// Array(3) [ 1, 2, 3 ]

还可以不完全结构,即等号左边只匹配右边的一部分

let [x, y] = [1, 2, 3];
x // 1
y // 2

let [a, [b], c] = [1, [2, 3], 4];
a // 1
b // 2
c // 4

如果右边不是数组(不可遍历)则会报错

# 字符串解构赋值

字符串可以进行解构赋值,因为字符串可以被转换为类似于数组的对象

const [a, b, c, d] = 'yian';
a // y
b // i 
c // a 
d // n

还可以对字符串的length属性进行解构赋值(数组的也可以)

let { length : len } = 'yian';
len // 4

# 函数参数的解构赋值

var userObj = { name: 'yian', age: 20, sex:'男', _id:'aaa' }

function logObj(name, age, sex, _id) {
    console.log(name, age, sex, _id);  
}
logObj(userObj.name, userObj.age, userObj.sex)
// yian 20 男 undefined

使用logObj函数的时候传递的参数实在是太麻烦了,你可以使用以下方法来简化.

function logObj( {name, age, sex, _id = 'bbb'} ) {
    console.log(name, age, sex, _id);  
}
logObj(userObj);
// yian 20 男 bbb

函数参数的解构也可以赋默认值

function add([a = 0, b = 0]){ return a + b; }
add([2, 5]) // 7

# 字符串新增方法

简单的学了几个,具体的还是需要看文档

  • String.fromCodePoint(): 用于从 Unicode 码点返回对应字符
  • String.raw(): 返回一个斜杠都被转义的字符串,大多用于模板字符串的处理
console.log( `Hi\n${2+3}!` ) 
// Hi
// 5!
console.log( String.raw(`Hi\n${2+3}!`) )  
// 'Hi\n5!'
  • codePointAt(): 正确处理 4 个字节储存的字符,返回一个字符的码点.
  • normalize()

之前学到的indexOf方法可以确定一个字符串是否包含在另一个字符串中,ES6新增了以下三个方法

  • includes: 表示是否找到该字符串,返回一个布尔值
  • startsWith: 表示参数字符串是不是在原字符串头部,返回一个布尔值
  • endsWith: 表示参数字符串是不是在原字符串尾部,返回一个布尔值

三个方法都支持第二个参数,表示搜索开始位置

let str = 'yianjun';

str.includes('yian')    // true
str.startsWith('y')     // true
str.endsWith('n')       // true

str.includes('yian', 2)    // false
str.startsWith('y', 0)     // true
str.endsWith('n', 4)       // true
  • 实例方法:repeat(),返回一个新的字符串,表示将原字符串重复N次
'an'.repeat(3) // ananan
  1. 参数如果是小数则会被向下取整'na'.repeat(2.9) // "nana"
  2. 参数在0到-1之间等同于零,返回空字符串
  3. 参数NaN等同于0
  4. 参数为负数或者Infinity会报错
  5. 参数是字符串会先转换为数字
'an'.repeat('yi') // ''       解释: +'yi' = NaN = 0
'an'.repeat('3')  // ananan
  • 实例方法padStart(),padEnd(): 用于补全字符串,如果字符串不够长则会在头部或尾部进行补全
  1. 第一个参数为补全后的最大长度,第二个是用来补全的字符串
'x'.padStart(6, 'ab') // ababax
  1. 原字符串长度大于等于最大长度返回原字符串
'xxx'.padStart(2, 'ab') // xxx
  1. 补全字符串加上原字符串后长度大于最大长度,则会将补全字符串截去超出的长度,如例一.
  2. 省略第二个参数,则默认使用空格补全

常见用途是为数值补全指定位数.下面代码生成 10 位的数值字符串

'1'.padStart(10, '0') // "0000000001"

另一个用途是提示字符串格式

'12'.padStart(10, 'YYYY-MM-DD') // "YYYY-MM-12"
  • trimStart(),trimEnd()消除头部或尾部空格
  • matchAll,返回一个包含所有匹配正则表达式的结果及分组捕获组的迭代器,参见JS基础-正则表达式
  • replaceAll(): 见名知意,替换所有,用法与replace()相同,但是当它的第一个为正则表达式时必须要带g修饰符,否则会报错,这一点以replace不同
'aaabbbccc'.replaceAll('b', 'B')
// aaaBBBccc

'aaabbbccc'.replaceAll('b', () => {
    return 'cc';
})
// aaaccccccccc

replaceAll的第二个参数也可以是一个函数,函数的返回值将替换掉第一个参数匹配到的文本

  • at()方法接受一个整数作为参数,返回参数指定位置的字符,支持负索引(即倒数的位置)
const str = 'hello';
str.at(1) // "e"
str.at(-1) // "o"

如果参数位置超过字符串范围则返回undefined

# 函数参数的默认值

以上例子中我们给的参数少了一个,函数会将缺少的参数默认为undefined.所以返回结果为NaN

function fn(a, b, c){ console.log(a + b + c); }

fn(1, 2)  // 1 + 2 + undefined = NaN

在ES6后,ES6 允许为函数的参数设置默认值,即直接写在参数定义的后面,一般为尾部参数设置默认值,如果为非尾部参数设置默认值,实际上这个参数是没法省略的

function fn(a, b, c = 0){ console.log(a + b + c); }

fn(1, 2) // 1 + 2 + 0 = 3

# rest参数

ES6引入rest参数...变量名,用于获取函数的多余参数,这样就不需要使用arguments对象了,rest参数搭配的变量是一个数组,该变量将多余参数放入数组中

function Add(...values) {
    let a = values.map( x => x * 2);
    return a;
}
Add(1, 3, 4, 5);   // [ 2, 6, 8, 10 ]

arguments对象不是数组,是一个类数组.如果使用数组的方法需要使用Array.from()先转化为数组,而rest参数是一个真正的数组

rest 参数之后不能再有其他参数(即只能是最后一个参数)否则会报错

function fn(a, ...values, b){ ... } // 报错

"use strict";  // 整个脚本都开启严格模式的语法
let v = "Hi!  I'm a strict mode script!";

# 箭头函数

箭头函数表达式的语法比函数表达式更简洁,并且没有自己的this,arguments,super或new.target.箭头函数表达式更适用于那些本来需要匿名函数的地方,并且它不能用作构造函数.

var fn = (a, b) => { console.log(a + b); }
fn(10, 20);

var fn = function(a, b){ console.log(a + b); }
fn(10, 20);

当函数的参数只有一个时小括号也可以简写

var fn2 = param => { console.log(param); }

当函数语句只有一条时,可以再次简写(不要return)

var fn3 = param => param.toString()

var fn3 = function(param) { return param.toString(); }

注意

  • 在箭头函数中没有自己的this指向, 上级作用域的this指向谁,箭头函数中的this就指向谁
  • 不可以当做构造函数.也就是说不能使用new命令
  • 不可以使用arguments对象,该对象在函数体内不存在,需要则要用rest参数代替
var obj = {
    fn: ()=>{ console.log(this); },
    fn1: function(){ console.log(this); }
}
obj.fn();   // Window

obj.fn1();  // Object { fn: fn(), fn1: fn1() }

# Proxy

Proxy(代理器)用于修改某些操作的默认行为,如下案例,Proxy在目标对象之前架设了一层拦截,外界对该对象的访问,都必须先通过这层拦截,因此提供了一种机制,可以对外界的访问进行过滤和改写.

    name: 'yian',
    age:18
}

    get: function(target, key){
        if(key === 'name'){
            if(target[key] === '易安'){
                return target[key] = 'YiAn';
            } else {
                return '该用户无权限';
            }
        }
    }
})

p.name = '易安';
console.log('obj: ', obj);
console.log('p.name: ', p.name);
// obj:  { name: '易安', age: 18 }
// p.name:  YiAn


p.name = 'aaa';
console.log('obj: ', obj);
console.log('p.name: ', p.name);
// obj:  { name: 'aaa', age: 18 }
// p.name:  该用户无权限
  • get方法用于拦截某个属性的读取操作,可以接受三个参数,依次为目标对象、属性名和 proxy 实例本身(严格地说,是操作行为所针对的对象),其中最后一个参数可选
  • set方法用来拦截某个属性的赋值操作,可以接受四个参数,依次为目标对象、属性名、属性值和 Proxy 实例本身,其中最后一个参数可选
// 实现数组读取负数索引
function proxyArray (target) {
    // 定制拦截行为
    let handler = {
        get(target, propKey, receiver) {
            // 转化为Number类型
            let index = Number(propKey);
            if(index < 0) {
                propKey = String(target.length + index);
            }
            // 每一个Proxy对象的拦截操作,内部都调用对应的Reflect方法，保证原生行为能够正常执行
            return Reflect.get(target, propKey, receiver);
        }
    };
    // 生成Proxy实例
    return new Proxy(target, handler);
}

    }

  • reject函数将Promise对象的状态从未完成变成已失败,在异步操作失败时调用
  • Promise实例生成以后,可以用then方法分别指定resolved状态和rejected状态的回调函数

# iterator和for……of循环

JS原有表示集合的数据结构主要为Array和Object，ES6新增了Map和Set。Iterator就是一个统一的接口机制，来处理所有不同的数据结构。任何数据结构只要部署Iterator接口就能完成遍历操作。

  • 默认的 Iterator 接口部署在数据结构的Symbol.iterator属性，只要具有这是属性就是可遍历的
  • 原生具备Iterator接口的数据结构为：Array、Map、Set、String、TypedArray、函数的arguments对象、NodeList对象
  • 调用Iterator接口的场合
    • 解构赋值
    • 扩展运算符
    • yield*
    • 任何接收数组作为参数的场合

# for……of

遍历所有数据结构的统一的方法，一个数据结构只要部署了Symbol.iterator属性，就被视为具有 iterator 接口，就可以用for...of循环遍历它的成员。也就是说，for...of循环内部调用的是数据结构的Symbol.iterator方法。

# Generator函数

Generator 函数是 ES6 提供的一种异步编程解决方案.function关键字和函数名之间有个星号,yield表达式定义不同的内部状态

function* hello() {
    yield 'hello';
    yield 'world';
    return 'ending';
}

var a = hello();

Generator函数调用方法与普通函数一样,但是,调用Generator函数后,函数并不执行,必须使用遍历对象的next方法使指针状态移向下一个状态.也就是说每次调用next,函数就从头一次的停止位置执行到遇到下一个yield表达式或return语句为止,Generator函数是分段执行的,yield表达式是暂停的标记

a.next()    // Object { value: "hello", done: false }
a.next()    // Object { value: "world", done: false }
a.next()    // Object { value: "ending", done: true }
a.next()    // Object { value: undefined, done: true }

# next的参数

function* f(){
    for(var i = 0; true; i++){
        var reset = yield i;
        if(reset){ 
            i = -1;
            console.log('12222');    
        }
    }
}
var f = f();
f.next();      // {value: 0, done: false}
f.next();      // {value: 1, done: false}
f.next(true);  // {value: 0, done: false}
f.next();      // {value: 1, done: false}
f.next();      // {value: 2, done: false}

# for……of

for...of循环可以自动遍历 Generator 函数运行时生成的Iterator对象,且此时不再需要调用next方法

function* hello() {
    yield '1';
    yield '2';
    return '3';
}
for (let i of hello()){
    console.log(i);
}
// 1 2

一旦next方法的返回对象的done属性为true,for...of循环就会中止,且不包含该返回对象

# async函数对 Generator 函数的改进

  1. 内置执行器

Generator函数执行需要执行器,所有才有了co模块,而async函数自带执行器,与普通函数执行相同

  1. 更好的语义

async和await,比起星号和yield,语义更清楚了

  1. 更广的适用性

yield命令后面只能是 Thunk 函数或 Promise 对象,而async函数的await命令后面,可以是 Promise 对象和原始类型的值(数值、字符串和布尔值,但这时会自动转成立即 resolved 的 Promise 对象)

  1. 返回值是Promise

async函数的返回值是 Promise 对象,Generator函数返回的是遍历器

# 块级作用域

  • 建议不再使用var命令,而是使用let命令取代.

  • 使用数组成员对变量赋值时,优先使用解构赋值
const arr = [1, 2, 3, 4];

const a = arr[0];
const b = arr[1];

// good 
const [a, b] = arr;
  • 函数的参数如果是对象的成员,优先使用解构赋值
// bad
function userName(user) {
  const firstName = user.firstName;
  const lastName = user.lastName;
}

// good
function userName(obj) {
  const { firstName, lastName } = obj;
}

// best
function userName({ firstName, lastName }) {
}
  • 如果函数返回多个值,优先使用对象的解构赋值,而不是数组的解构赋值.这样便于以后添加返回值,以及更改返回值的顺序
// bad
function processInput(input) {
  return [left, right, top, bottom];
}

// good
function processInput(input) {
  return { left, right, top, bottom };
}

const { left, right } = processInput(input);
