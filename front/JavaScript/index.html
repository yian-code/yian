原型-原型链

  • 在每一个对象上都有一个prototype属性,这个属性是专门用于保存所有该对象下实例所共享的属性和方法,这个prototype属性就是我们所谓的原型对象
  • 在每一个实例上都有一个__proto__属性,用于查找该实例的原型对象
  • 原型链:在每一个实例对象上都有一个__proto__的属性,该属性指向所在的原型,当对象上调用某个没有的方法时,就会通过__proto__去找原型对象查看,如果原型对象也没有该方法,则继续通过__proto__向上查找,一直找到null为止,这种层层查找的方式,被称之为原型链
function Person(){}
var user = new Person();

console.log(Person.prototype == user.__proto__)  // true
  1. 组合法创建对象(构造+原型)
function Person(name,age){ // 不同属性用构造
	this.name = name;
	this.age = age;
}
Person.prototype.say = function(){ // 相同方法用原型
	console.log("共享的方法");
}
var user1 = new Person("易安",18);
var user2 = new Person("yian",20);
user1.say();
user2.say();  // 两个对象调用了同一个say方法

# Object.entries()

Object.entries()将对象按照键值对key:value形式转换为二维数组,对象有多少个属性,数组的长度就是多少.

var User = {
	name: 'yian',
	age: '20',
	sex: '1',
	fn:function(){
		console.log("YiAn")
	}
}
var arr = Object.entries(User);
console.log(arr);
// (4) [Array(2), Array(2), Array(2), Array(2)]
// 0:(2) ['name', 'yian']
// 1:(2) ['age', 90]
// 2:(2) ['sex', '1']
// 3:(2) ['fn', ƒ]

如果需要转换为一维数组则可以使用数组的flat()方法

var arr1 = Object.entries(User).flat();
console.log(arr1);
// (8) ['name', 'yian', 'age', '20', 'sex', '1', 'fn', ƒ]
 


# Object.freeze()

Object.freeze(obj)方法可以冻结一个对象.

  • 一个被冻结的对象再也不能被修改.
  • 冻结了一个对象则不能向这个对象添加新的属性,不能删除已有属性,不能修改该对象已有属性的可枚举性,可配置性,可写性,以及不能修改已有属性的值.
  • 冻结一个对象后该对象的原型也不能被修改.
  • freeze() 返回和传入的参数相同的对象.
var freObj = Object.freeze(User);
console.log(freObj); // {name: 'yian', age: '20', sex: '1', fn: ƒ}   
var delObj = delete(User.age);
console.log(delObj); // false
console.log(User); // {name: 'yian', age: '20', sex: '1', fn: ƒ}

# Object.defineProperty()

在一个对象上定义一个新属性,或者修改一个对象的现有属性,并返回此对象Object.defineProperty(object,prop,descript)

  • object:  要添加或者修改属性的目标对象.
  • prop:  要定义或修改属性的名称.
  • descript:  要定义或修改的属性描述符.
function Person(name, age, id) {
    this.name = name,
    this.age = age,
    this.id = id
}

var userObj = new Person("yian", 18, 413026);
Object.defineProperty(userObj, "id", {
    configurable: true, // 是否可以删除 默认 false
    enumerable: true, // 是否可以枚举 默认 false
    // value:12, // 设置该属性的值 默认 undefined
    // writable:true, // 是否可以修改 默认 false
    get: function() {
        console.log("get触发");
        return -1;
    },
    set: function(e) { //有个形参,接收的是修改的值
        console.log("new value hello");
    }
})

userObj.id = 12222;
userObj.id = 122;

# 响应式基础

注意保存原始页面:因为每次都需要正则匹配,如果不保存,修改后下一次就找不到正则匹配目标,也就无法达到响应式效果

<body>
    <p>{{data.message}}</p>
    <button>{{data.message}}</button>
    <span>{{data.message}}</span>
    <script>
        var body = document.body; // 获取整个页面
        var htmlStr = body.innerHTML;  //保存原始页面
        var data = {
            message : "hello"
        } 
		// 函数封装,传入对象及需要修改的属性名
        function setDataPro(obj,proName){ 
			// 创建了一个空对象 des
            var des = Object.create(obj); 
			// 为空对象设置set方法
            des.set = function(val){ 
				// 正则表达式匹配 {{data.message}}
                var reg = /\{\{data\.message\}\}/g; 
                // 每次重新设置后都在原始页面基础上替换,并将修改后的页面内容回填入body
                body.innerHTML = htmlStr.replace(reg,val);
            }
            Object.defineProperty(obj,proName,des); 
        }
		// 函数调用,修改了data对象的message属性
        setDataPro(data,"message"); 
		// 修改页面内容只需要修改message属性值即可
        data.message = "你好"; 
    </script>
</body>

function randomFn(){  // 封装随机四位验证码的函数
	var randomStr = ""; //创建一个空字符串 用于接收四位随机字符
	for(var i = 0 ; i < 4 ; i++){ //循环四次
	//随机一位字符串中的下标

# substring

str.substring(indexStart[, indexEnd])

  • indexStart:需要截取的第一个字符的索引,该索引位置的字符作为返回的字符串的首字母.
  • indexEnd:可选.一个 0 到字符串长度之间的整数,以该数字为索引的字符不包含在截取的字符串内.
var anyString = "Mozilla";
console.log(anyString.substring(0,6)) // Mozill

 

注意:JS中截取包括起始下标位置,不包括下标结束位置

  1. 如果indexStart等于indexEnd.substring 返回一个空字符串
var anyString = "Mozilla";
console.log(anyString.substring(4,4)) // ""
  1. 如果省略indexEnd.substring提取字符一直到字符串末尾
var anyString = "Mozilla";
console.log(anyString.substring(2)) // zilla
  1. 如果任一参数小于0或为NaN.则被当作0
var anyString = "Mozilla";
console.log(anyString.substring(-2,3)); // Moz
console.log(anyString.substring(NaN,3)); // Moz
  1. 如果任一参数大于stringName.length则被当作stringName.length
console.log(anyString.substring(0,7));  // Mozilla
console.log(anyString.substring(0,10)); // Mozilla
  1. 如果indexStart大于indexEnd则substring的执行效果就像两个参数调换了一样
console.log(anyString.substring(4,7)); // lla
console.log(anyString.substring(7,4)); // lla
# slice

slice()方法提取某个字符串的一部分,并返回一个新的字符串,且不会改动原字符串.

str.slice(beginIndex[, endIndex])
  • beginIndex:从该索引(以0为基数)处开始提取原字符串中的字符.如果值为负数,会被当做strLength + beginIndex 看待,这里的strLength是字符串的长度(例如,如果beginIndex是-3则看作是strLength - 3）
  • endIndex:可选,在该索引(以0为基数)处结束提取字符串.如果省略该参数,slice()会一直提取到字符串末尾.如果该参数为负数,则被看作是strLength + endIndex这里的strLength就是字符串的长度(例如,如果endIndex是-3,则是 strLength - 3)
  • slice不会修改原字符串(只会返回一个包含了原字符串中部分字符的新字符串)

slice()提取的新字符串包括beginIndex但不包括endIndex

	// 1. 将？所在的位置获取到
	var index = url.indexOf("?");
	// 2. 将？后面的内容截取出来
	var str = url.slice(index + 1);

	// 4. 使用等号再转换为字符串
	var newStr = arr.join("=");

	// 5. 再使用等号作为分隔符转换为数组
	var newArr = newStr.split("=");
	console.log(newArr) //['id', '1', 'city', '北京', 'data_in', '2021/09/26', 'data_out', '2021/09/28']
	// 6. 创建对象
	var  obj = {};
	// 7. 将数组的奇数项作为对象的属性名,偶数项作为对象的属性值来添加
	for(var i = 0 ; i < newArr.length ; i = i + 2){
		obj[newArr[i]] = newArr[i + 1] ;  
	}
	// 8. 将对象返回到函数调用处
	return obj
}

# 正则表达式中的特殊字符

表达式描述
[abc]查找方括号之间的任何字符,包括转义序列,可以使用破折号-来指定一个字符范围.例如,[abcd]和[a-d]是一样的.他们都匹配"brisket"中的‘b’,也都匹配“city”中的‘c’
[^abc]一个反向字符集, 它匹配任何没有包含在方括号中的字符
[0-9]查找任何从0至9的数字.[]中只匹配一个,匹配多个需要加{}字符来限制,例如[0-9]{1,3}
x|y匹配x或者y ,等价于[xy]
()小括号()中的内容被作为单个字符来匹配
{n}n是一个正整数,匹配了前面一个字符刚好出现了n次
{n,}n是一个正整数,匹配前一个字符至少出现了n次
{n,m}n和m都是整数.匹配前面的字符至少n次.最多m次.如果n或者m的值是0.这个值被忽略
^匹配输入的开始.如果多行标志被设置为 true.那么也匹配换行符后紧跟的位置
例如./^A/并不会匹配"an A"中的'A',但是会匹配"An E"中的'A'
$匹配输入的结束.如果多行标志被设置为 true.那么也匹配换行符前的位置
例如:/t$/并不会匹配"eater"中的't',但是会匹配"eat" 中的 't'
.默认匹配除换行符之外的任何单个字符
n+匹配前面一个表达式 1 次或者多次.等价于 {1,}
n*匹配前一个表达式 0 次或多次.等价于{0,}
n?匹配前面一个表达式 0 次或者 1 次.等价于 {0,1}
\d匹配一个数字.等价于[0-9]
\D匹配一个非数字字符.等价于[^0-9]
\w匹配一个单字字符(字母,数字或者下划线).等价于 [A-Za-z0-9_]
\W匹配一个非单字字符.等价于 [^A-Za-z0-9_]
\s查找空白字符,包括空格、制表符、换页符和换行符等价于
[ \f\n\r\t\v\u00a0\u1680\u180e\u2000-\u200a\u2028\u2029\u202f\u205f\u3000\ufeff]
\S等价于[ \f\n\r\t\v\u00a0\u1680\u180e\u2000-\u200a\u2028\u2029\u202f\u205f\u3000\ufeff]
\b匹配单词边界 /\bm/匹配moon中的m
\B匹配一个非单词边界
\uxx匹配一个两位十六进制数(\x00-\xFF)表示的字符
\uxxxx查找以十六进制数xxxx规定的Unicode字符

强烈建议不要使用for…in来遍历数组,for-in遍历的是对象的key或者是数组的下标,当然最重要的原因是,他会到原型链上寻找,遍历其中新增加的属性的key值

for (var index in arr){
	console.log(index);  // typeof(index) = String 
	console.log(arr[index]);
}

# 添加与删除元素

  • push:将一个或多个元素添加到数组的末尾,并返回该数组的新长度.
var arr = ['html',"css","javaScript"];
// 在数组的末尾追加新的数组项 push()
var a = arr.push("red","green","blue")
console.log(arr)
// Array(6) [ "html", "css", "javaScript", "red", "green", "blue" ]
console.log(a)  // a = 6

 




  • pop:方法从数组中删除最后一个元素,并返回该元素的值.此方法会更改数组的长度.
var arr = ['html',"css","javaScript"];
var b = arr.pop();
console.log(arr);
// Array [ "html", "css" ]
console.log(b);  // b = javaScript
  • unshift:方法将一个或多个元素添加到数组的开头,并返回该数组的新长度(该方法修改原有数组)
var c = arr.unshift("yellow",'pink');
console.log(arr);
// Array(5) [ "yellow", "pink", "html", "css", "javaScript" ]
console.log(c)  // 5
  • shift:从数组中删除第一个元素,并返回该元素的值.此方法更改数组的长度.
var arr = ['html',"css","javaScript"];
var d = arr.shift();
console.log(arr);
// Array [ "css", "javaScript" ]
console.log(d)  
//  html

# 反转数组

reverse()方法将数组中元素的位置颠倒,并返回该数组.数组的第一个元素会变成最后一个,数组的最后一个元素变成第一个.该方法会改变原数组.

var a = [1, 2, 3];
var val = a.reverse();
console.log(val) //  [3, 2, 1]
console.log(a); // [3, 2, 1]



 

# 数组排序

sort()方法用原地算法对数组的元素进行排序,并返回数组.默认排序顺序是在将元素转换为字符串,然后比较它们的UTF-16代码单元值序列时构建的.

# 截取数组

# slice

arr.slice([begin[, end]]):返回一个新的数组对象,这一对象是一个由begin和end决定的原数组的浅拷贝(包括 begin，不包括end).原始数组不会被改变

  • begin: 如果该参数为负数,则表示从原数组中的倒数第几个元素开始提取,slice(-2)表示提取原数组中的倒数第二个元素到最后一个元素(包含最后一个元素)如果省略begin则slice从索引0开始.如果begin超出原数的索引范围,则会返回空数组.
  • end: 在该索引处结束提取原数组元素,如果该参数为负数,则它表示在原数组中的倒数第几个元素结束抽取,不包含最后一个元素,如果end被省略.则 slice会一直提取到原数组末尾.如果end大于数组的长度,slice 也会一直提取到原数组末尾.
var arr = ["a1","a2",'a3',"a4","a5"];
var val1 = arr.slice(1,3);
console.log(val1); // Array [ "a2", "a3" ]

var val2 = arr.slice(-3);
console.log(val2) // Array(3) [ "a3", "a4", "a5" ]

var val3 = arr.slice(3);
console.log(val3) // Array [ "a4", "a5" ]

var val4 = arr.slice();
console.log(val4) // Array(5) [ "a1", "a2", "a3", "a4", "a5" ]

var val5 = arr.slice(-3,-1);
console.log(val5) // Array [ "a3", "a4" ]

var val6 = arr.slice(5);
console.log(val6) // Array [  ]




 








 




# splice

array.splice(start[, deleteCount[, item1[, item2[, ...]]]])通过删除或替换现有元素或者原地添加新的元素来修改数组,并以数组形式返回被修改的内容.此方法会改变原数组.
splice的返回值:由被删除的元素组成的一个数组.如果只删除了一个元素.则返回只包含一个元素的数组,如果没有删除元素.则返回空数组.

  • 删除操作
var arr = ["a","b",'c',"d","e"];
var arr1 = arr.splice(3,1); //表示从下标3的位置截取一个 
console.log(arr1);// Array [ "d" ]
console.log(arr); // Array(4) [ "a", "b", "c", "e" ]
  • 修改操作
var arr = ["a","b",'c',"d","e"];
var arr1 = arr.splice(3,1,"vue"); //表示从下标3的位置截取一个,并添加一个新内容
console.log(arr1);// Array [ "d" ]
console.log(arr); // Array(5) [ "a", "b", "c", "vue", "e" ]
  • 添加操作
var arr = ["a","b",'c',"d","e"];
var arr1 = arr.splice(3,0,"vue"); //表示从下标3的位置一个都不截取,并添加一个新内容
console.log(arr1);// Array [] 没有删除元素,则返回空数组
console.log(arr); // Array(6) [ "a", "b", "c", "vue", "d", "e" ]

# 数组位置的查找

# indexOf

从数组的前面向后查找:indexOf()方法返回在数组中可以找到一个给定元素的第一个索引,如果不存在,则返回-1arr.indexOf(searchElement[, fromIndex])

  • searchElement:要查找的元素
  • fromIndex(可选):开始查找的位置.如果参数中提供的索引值是一个负值,则将其作为数组末尾的一个抵消,即-1表示从最后一个元素开始查找,-2表示从倒数第二个元素开始查找.并不改变其查找顺序,查找顺序仍然是从前向后查询数组,如果抵消后的索引值仍小于0,则整个数组都将会被查询.其默认值为0
var arr = ["a","b",'c',"d","e"];
var idx = arr.indexOf("c")
console.log(idx) // 2 

var idx2 = arr.indexOf("c",3)
console.log(idx2) // -1

var idx3 = arr.indexOf("c",-3)
console.log(idx3) // 2




 


 

# lastIndexOf

lastIndexOf()方法返回指定元素(也即有效的JavaScript值或变量)在数组中的最后一个的索引,如果不存在则返回 -1.从数组的后面向前查找,从fromIndex处开始.默认起始位置是数组的最后一项,刚好与indexOf顺序相反

  • fromIndex如果为负值,将其视为从数组末尾向前的偏移.即使该值为负.数组仍然会被从后向前查找
var arr = ["a","b",'c',"d","e"];
var idx = arr.lastIndexOf("c")
console.log(idx) // 2

var idx2 = arr.lastIndexOf("c",1)
console.log(idx2) // -1

var idx3 = arr.lastIndexOf("c",-4)
console.log(idx3) // -1

# some

查询数组中是否存在满足某些条件的数组项,满足则返回true 不满足返回false.语法:arr.some(callback(element[, index[, array]])[, thisArg])

  • callback: 用来测试每个元素的函数,接受三个参数:
    • element:  数组中正在处理的元素
    • index:可选  数组中正在处理的元素的索引值
    • array:可选 some()被调用的数组
  • thisArg可选.执行callback时使用的this值

返回值(Boolean):数组中有至少一个元素通过回调函数的测试就会返回true.所有元素都没有通过回调函数的测试返回值才会为false.

var arr = ['1', '3', '4'];
var a = arr.some(function(item,index,array){  
	return item < 3 //判断数组中有没有小于3的
})
// var a = arr.some(x => x < 3)
console.log(a) // true

# every

查询数组项中的所有数组项是否全部满足条件,如果全都满足返回true 只要有一个不满足 返回的就是false.语法:arr.every(callback(element[, index[, array]])[, thisArg])

  • callback: 用来测试每个元素的函数,接受三个参数:
    • element:  数组中正在处理的元素
    • index:可选  数组中正在处理的元素的索引值
    • array:可选 every()被调用的数组
  • thisArg:执行callback时使用的this值

如果回调函数的每一次返回都为truthy值,返回true.否则返回 false

var arr = ['1', '3', '4'];
var b = arr.every(function(item){
	return item >= 3 // arr中全都大于3吗
})
// var b = arr.every(x => x >= 3)
console.log(b) // false

# map

返回一个全新的数组,这个新数组由原数组中的每个元素都调用一次提供的函数后的返回值组成,常用于数组中的内容批量处理.详细参见MDNopen in new window

  • callback: 生成新数组元素的函数,接受三个参数:
    • element:  数组中正在处理的当前元素
    • index:可选  数组中正在处理的当前元素的索引值
    • array:可选 map()被调用的数组
  • thisArg可选.执行callback时使用的this值

返回一个由原数组每个元素执行回调函数的结果组成的新数组

var arr = [1,2,3,4,5];
var arr2 = arr.map(function(item,index,array){
	return  item * 2

# filter

filter()方法创建一个新数组, 其包含通过所提供函数实现的测试的所有元素var newArray = arr.filter(callback(element[, index[, array]])[, thisArg])

  • callback: 用来测试数组的每个元素的函数.返回true表示该元素通过测试,保留该元素,false则不保留,接受三个参数:
    • element:  数组中正在处理的元素
    • index:可选  数组中正在处理的元素的索引值
    • array:可选 调用了filter的数组本身
  • thisArg可选.执行callback时使用的this值

返回一个新的、由通过测试的元素组成的数组,如果没有任何数组元素通过测试,则返回空数组

var arr = ['1', '3', '4','5','7','9'];
var newArr = arr.filter(function(item){
	return item > 4;  
})
// var newArr = arr.filter(item => item > 4)
console.log(newArr)
// Array(4) [ "4", "5", "7", "9" ]

# 数组去重

  1. 借助indexOf方法:判断此元素在该数组中首次出现的位置下标与循环的下标是否相等
  • 把去重方法写在Array对象上,参考原型食用:
var arr2 = [1,1,3,5,6,4,8,5,3,2,1,1,1];

Array.prototype.deWight = function(){
	for (let i = 0; i < this.length; i++) {
		if(this.indexOf(this[i],i + 1) !== -1){
			this.splice(i, 1); // 删除数组元素后长度减1,后面的元素前移
			i--; // 上一步删除了一个,自减一,这样才能从头开始继续去重
// 去重过程如下: console.log(this);   
			// [1, 3, 5, 6, 4, 8, 5, 3, 2, 1, 1, 1]
			// [3, 5, 6, 4, 8, 5, 3, 2, 1, 1, 1]
			// [5, 6, 4, 8, 5, 3, 2, 1, 1, 1]
			// [6, 4, 8, 5, 3, 2, 1, 1, 1]
			// [6, 4, 8, 5, 3, 2, 1, 1]
			// [6, 4, 8, 5, 3, 2, 1]
		}   
	}
}
var newArr = arr2.deWight();   
console.log(newArr)  // [6, 4, 8, 5, 3, 2, 1]
  • 封装函数方式:更容易理解
var arr = [1, 1, 3, 5, 6, 4, 8, 5, 3, 2, 1, 1, 1];

function deweight(arr) {
  for (var i = 0; i < arr.length; i++) {
    if (arr.indexOf(arr[i]) != i) {
      arr.splice(i, 1);
      i--;
    }
  }
  return arr;
}
var newArr = deweight(arr);
console.log(newArr);
// Array(7) [ 1, 3, 5, 6, 4, 8, 2 ]
