# 图片的预加载和懒加载

前端性能优化之图片

  • 精灵图(雪碧图) 将图片转换为base64字符串格式使用 使用svg图标
  • 懒加载 : 页面一加载的时候,不会将所有的图片都加载完毕,而是当用户需要查看该图片时,再进行加载
  • 预加载 : 将图片从服务器提前获取到,然后存入本地缓存中,之后用户查看图片的时候,就不需要从服务器获取,而是直接从本地缓存中读取,这样读取的速度会更快,更方便轮播图可能会使用
// 预加载
var imgUrl = ["./imgs/1.png","./imgs/2.png","./imgs/3.png","./imgs/4.png"]
var img = new Image();
console.log(img)
img.src = imgUrl[0];
img.onload = function(){	
    // 图片加载完再执行的操作
}

# 同步异步

需求 ：0 1 2 3 4

//闭包
for (var i = 0; i < 5; i++) {
    (function(i) {
        setTimeout(function() {
            console.log(i); //5个值
            //内部 没有i 上一层作用域找--->函数a--->有参数i -->使用i
        })
    })(i);
}

//使用let 块级作用域解决
for (let i = 0; i < 5; i++) {
    setTimeout(function() {
        console.log(i); //5个值
    })
}

# 多维数组降维(递归)

// 接收降维后的数组
var newArr = [];

// 递归函数
function fn(arr) {
    arr.forEach(item => {
        if(Array.isArray(item)) {
            fn(item);
            // arguments.callee(item);  // 与fn(item)等同
        } else {
            newArr.push(item);
        }
    });
    return newArr;
}
let  arr = [1, 2, [3, 4, [5, 6, [7, 8, [9] ] ] ] ]

console.log(fn(arr))
// [1, 2, 3, 4, 5, 6, 7, 8, 9]
